# apps/api/v1/routes.txt
#
# V1 uses controller-level auth via authorized(true|false), NOT Otto's
# route-level auth strategies. Auth metadata uses openapi_auth= (not auth=)
# so Otto's RouteAuthWrapper doesn't try to enforce unregistered strategies.
# The OpenAPI parser reads openapi_auth= for schema generation.
#
# Other attributes (content=, response=, deprecated=) are declarative
# metadata for the otto-route-parser used in OpenAPI schema generation.
# V1 uses response=default so Otto skips its JSONHandler pipeline (V1
# controllers serialize responses directly). This has no effect on the
# OpenAPI spec because the generator resolves schemas via the SCHEMAS
# constant in the controller class, not the response= attribute.

GET    /status                               V1::Controllers::Index#status      openapi_auth=basic,anonymous response=default
GET    /authcheck                            V1::Controllers::Index#authcheck   openapi_auth=basic response=default

POST   /share                                V1::Controllers::Index#share       openapi_auth=basic,anonymous content=form response=default
POST   /generate                             V1::Controllers::Index#generate    openapi_auth=basic,anonymous content=form response=default

POST   /create                               V1::Controllers::Index#create      openapi_auth=basic,anonymous content=form response=default

POST   /secret/:key                          V1::Controllers::Index#show_secret openapi_auth=basic,anonymous content=form response=default sensitive=true

# Canonical receipt paths
# NOTE: Specific sub-paths (burn) must precede the :key wildcard so
# Otto's first-match-wins dynamic routing resolves them correctly.
GET    /receipt/recent                       V1::Controllers::Index#show_receipt_recent openapi_auth=basic response=default
POST   /receipt/:key/burn                    V1::Controllers::Index#burn_secret  openapi_auth=basic,anonymous content=form response=default sensitive=true
GET    /receipt/:key                         V1::Controllers::Index#show_receipt openapi_auth=basic,anonymous response=default sensitive=true
POST   /receipt/:key                         V1::Controllers::Index#show_receipt openapi_auth=basic,anonymous content=form response=default sensitive=true
